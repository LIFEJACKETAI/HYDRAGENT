    87→function formatAppointmentDate(dateStr: string): string {
    88→  const date = new Date(dateStr)
    89→  let dayLabel = ''
    90→  if (isToday(date)) dayLabel = 'Today'
    91→  else if (isTomorrow(date)) dayLabel = 'Tomorrow'
    92→
    93→  const formatted = format(date, 'EEE, MMM d')
    94→  const time = format(date, 'h:mm a')
    95→  if (dayLabel) return `${dayLabel} at ${time}`
    96→  return `${formatted} at ${time}`
    97→}
